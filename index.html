N Queen Problem | Return all Distinct Solutions to the N-Queens Puzzle


The n-queens is the problem of placing n queens on n Ã— n chessboard such that no two queens can attack each other. Given an integer n, return all distinct solutions to the n -queens puzzle. Each solution contains a distinct boards configuration of the queen's placement.


Code Implementation


// JavaScript code for N-Queens problem
function solveNQueens(n) {
    let board = Array(n).fill().map(() => Array(n).fill('.'));
    let result = [];
    
    function isSafe(row, col) {
        for (let i = 0; i < row; i++) {
            if (board[i][col] === 'Q') return false;
            if (col - (row - i) >= 0 && board[i][col - (row - i)] === 'Q') return false;
            if (col + (row - i) < n && board[i][col + (row - i)] === 'Q') return false;
        }
        return true;
    }
    
    function placeQueen(row) {
        if (row === n) {
            result.push(board.map(r => r.join('')));
            return;
        }
        for (let col = 0; col < n; col++) {
            if (isSafe(row, col)) {
                board[row][col] = 'Q';
                placeQueen(row + 1);
                board[row][col] = '.';
            }
        }
    }
    
    placeQueen(0);
    return result;
}
